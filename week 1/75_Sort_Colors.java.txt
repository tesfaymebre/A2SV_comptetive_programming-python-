class Solution {
    public void sortColors(int[] nums) {
        for(int i = 0; i < nums.length - 1; i++){
            int temp;
            for(int j = i+1; j < nums.length; j++){
                if(nums[j] < nums[i]){
                    temp = nums[i];
                    nums[i] = nums[j];
                    nums[j] = temp;
                }
            }
        }
    }
}
